ON OFF ON SPDT switch wiring
                 Throw 1 ----- GND
                 Throw 2 ----- 3.3V
                 Common  ----- ADC pin
3.3V ----[ Rtop ]----+----[ Rbottom ]---- GND
                      |
                    ADC pin
	Common → ADC pin
	One end → GND
	Other end → 3.3 V
	Middle position = OFF → no connection → ADC only sees the resistor divider.
Resulting voltages
	Position 1 (to GND): ADC ≈ 0" V" (switch dominates, low impedance to GND)
	Middle (OFF): ADC ≈ 1.65" V" (pure divider)
	Position 3 (to 3.3 V): ADC ≈ 3.3" V" (switch dominates, low impedance to VCC)
Now you truly get three stable, well separated levels.

struct Entry {
    char key[32];
    char value[32];
};

Entry table[] = {
    { "SSID",     "MyWiFi" },
    { "Password", "SuperSecret" },
    { "Mode",     "Station" }
};

const size_t tableSize = sizeof(table) / sizeof(table[0]);

const char* getValueByKey(const char* key) {
    for (size_t i = 0; i < tableSize; i++) {
        if (strcmp(table[i].key, key) == 0) {
            return table[i].value;
        }
    }
    return nullptr;
}

bool setValueByKey(const char* key, const char* newValue) {
    for (size_t i = 0; i < tableSize; i++) {
        if (strcmp(table[i].key, key) == 0) {
            strncpy(table[i].value, newValue, sizeof(table[i].value) - 1);
            table[i].value[31] = '\0';
            return true;
        }
    }
    return false;
}

void printTable() {
    for (size_t i = 0; i < tableSize; i++) {
        Serial.print("Key: ");
        Serial.print(table[i].key);
        Serial.print("  Value: ");
        Serial.println(table[i].value);
    }
}

void setup() {
    Serial.begin(115200);

    printTable();

    Serial.println("\nChanging Mode...");
    setValueByKey("Mode", "AccessPoint");

    Serial.println("\nNew table:");
    printTable();

    Serial.println("\nLookup SSID:");
    const char* ssid = getValueByKey("SSID");
    if (ssid) Serial.println(ssid);
}

void loop() {}

Define a POD style class (or struct)
To store an object as a blob, it must be trivially copyable (no pointers, no virtual methods, no dynamic memory).

struct DeviceConfig {
    uint32_t version;
    int32_t threshold;
    float gain;
    bool enabled;
};

#include <Preferences.h>

Preferences prefs;

DeviceConfig config = {
    .version = 1,
    .threshold = 42,
    .gain = 0.75f,
    .enabled = true
};

void saveConfig(const DeviceConfig &cfg) {
    prefs.begin("app", false);  // RW mode

    prefs.putBytes("config", &cfg, sizeof(DeviceConfig));

    prefs.end();
}

bool loadConfig(DeviceConfig &cfg) {
    prefs.begin("app", true);  // read-only

    size_t size = prefs.getBytesLength("config");
    if (size != sizeof(DeviceConfig)) {
        prefs.end();
        return false;  // no valid config stored
    }

    prefs.getBytes("config", &cfg, sizeof(DeviceConfig));

    prefs.end();
    return true;
}

void setup() {
    Serial.begin(115200);

    DeviceConfig loaded;

    if (loadConfig(loaded)) {
        Serial.println("Loaded config:");
        Serial.printf("version=%u threshold=%d gain=%.2f enabled=%d\n",
                      loaded.version, loaded.threshold,
                      loaded.gain, loaded.enabled);
    } else {
        Serial.println("No valid config found, saving defaults...");
        saveConfig(config);
    }
}

void loop() {
