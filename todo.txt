Idea to reduce complexity of PCB:
- move setting params (like instel potmeters and dipswitches) to permanent flash memory
See https://docs.espressif.com/projects/arduino-esp32/en/latest/api/preferences.html
Example:
#include <Preferences.h>
Preferences preferences;
void setup() {
  Serial.begin(115200);
  // Open Preferences with namespace "counter" in RW mode
  preferences.begin("counter", false);
  // Read counter, default to 0 if not exists
  unsigned int counter = preferences.getUInt("counter", 0);
  counter++;
  Serial.printf("Current counter value: %u\n", counter);
  // Store the counter value
  preferences.putUInt("counter", counter);
  // Close the Preferences
  preferences.end();
}
void loop() {}
