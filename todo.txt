Define a POD style class (or struct)
To store an object as a blob, it must be trivially copyable (no pointers, no virtual methods, no dynamic memory).

struct DeviceConfig {
    uint32_t version;
    int32_t threshold;
    float gain;
    bool enabled;
};

#include <Preferences.h>

Preferences prefs;

DeviceConfig config = {
    .version = 1,
    .threshold = 42,
    .gain = 0.75f,
    .enabled = true
};

void saveConfig(const DeviceConfig &cfg) {
    prefs.begin("app", false);  // RW mode

    prefs.putBytes("config", &cfg, sizeof(DeviceConfig));

    prefs.end();
}

bool loadConfig(DeviceConfig &cfg) {
    prefs.begin("app", true);  // read-only

    size_t size = prefs.getBytesLength("config");
    if (size != sizeof(DeviceConfig)) {
        prefs.end();
        return false;  // no valid config stored
    }

    prefs.getBytes("config", &cfg, sizeof(DeviceConfig));

    prefs.end();
    return true;
}

void setup() {
    Serial.begin(115200);

    DeviceConfig loaded;

    if (loadConfig(loaded)) {
        Serial.println("Loaded config:");
        Serial.printf("version=%u threshold=%d gain=%.2f enabled=%d\n",
                      loaded.version, loaded.threshold,
                      loaded.gain, loaded.enabled);
    } else {
        Serial.println("No valid config found, saving defaults...");
        saveConfig(config);
    }
}

void loop() {
